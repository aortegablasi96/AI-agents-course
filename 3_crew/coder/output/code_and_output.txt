# Calculate the first 10,000 terms of the series: 1 - 1/3 + 1/5 - 1/7 + ...
total_sum = 0.0

for i in range(10000):
    # For even index, we add the term, for odd index we subtract
    term = 1 / (2 * i + 1)  # This gives 1, 1/3, 1/5, 1/7, ...
    if i % 2 == 0:  # Even index
        total_sum += term
    else:           # Odd index
        total_sum -= term

# Multiply the total by 4
final_result = total_sum * 4
print(final_result)
